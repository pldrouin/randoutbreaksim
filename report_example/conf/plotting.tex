\def\drawfunc#1 #2\endline{%
  \addpoint(#1,#2)%
}

\def\drawh#1 #2 #3\endline{%
    \addpoint(#1,#3)\addpoint(#2,#3)%
}

\def\drawhup#1 #2 #3\endline{%
    \addpoint(#1,#3)\addpoint(#2,#3)%
}

\def\drawhdown#1 #2 #3\endline{%
    \addpoint(#2,0)\addpoint(#1,0)%
}

\def\drawpoints#1 #2\endline{%
  \edef\x{#1}%
  \edef\y{#2}%
  \pladobjfunc%
}

\def\draweypoints#1 #2 #3\endline{%
  \edef\x{#1}%
  \edef\y{#2}%
  \edef\eym{#3}%
  \edef\eyp{#3}%
  \pladobjfunc%
}

\def\draweypointsyave#1 #2 #3\endline{%

  \FPmul\y{\y}{0.5}%
  \FPsub\eyp{#3}{\y}%
  \edef\eym{\eyp}%
  \pladobjfunc%
}
