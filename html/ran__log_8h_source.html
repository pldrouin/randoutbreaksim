24  * @brief Finite logarithmic deviate.
25  *
26  * Returns a logarithmic deviate. Returned value is finite, and can be cast to
27  * the desired integer data type. Modified algorithm from "Non-Uniform Random Variate Generation
28 ", by Luc Devroye.
29  *
30  * @param rl handle
31  * @return finite logarithmic deviate
32  */
33 #define ran_log_finite_gen(rl) \
34 {\
35  double v=rng_rand_pu01(rl->s); /*We want a finite returned value;*/ \
36  \
37  if (v>=rl->p) return 1; \
38  \
39  else { \
40  double q=1-exp(rl->r*rng_rand_pu01(rl->s)); \
41  \
42  if (v<=q*q) return 1+log(v)/log(q); \
43  \
44  else if (v <= q) return 2; \
45  \
46  else return 1 ; \
47  } \
48 }
