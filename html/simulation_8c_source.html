13 int sim_pars_check(sim_pars const* pars)
14 {
15  int ret=0;
16 
17  if(pars->tbar<=0) {
18  fprintf(stderr,"%s: Error: tbar must be greater than 0\n",__func__);
19  ret-=1;
20  }
21 
22  if(pars->p<0) {
23  fprintf(stderr,"%s: Error: p must be non-negative\n",__func__);
24  ret-=2;
25  }
26 
27  if(pars->lambda<=0) {
28  fprintf(stderr,"%s: Error: lambda must be greater than 0\n",__func__);
29  ret-=4;
30  }
31 
32  if(pars->kappa<=0) {
33  fprintf(stderr,"%s: Error: kappa must be greater than 0\n",__func__);
34  ret-=8;
35  }
36 
37  if(pars->lbar<0) {
38  fprintf(stderr,"%s: Error: lbar must be non-negative\n",__func__);
39  ret-=16;
40 
41  } else if(pars->lbar>0 && pars->kappal<=0) {
42  fprintf(stderr,"%s: Error: kappal must be greater than 0 if lbar>0\n",__func__);
43  ret-=32;
44  }
45 
46  if(pars->q<0) {
47  fprintf(stderr,"%s: Error: q must be non-negative\n",__func__);

50  } else if(pars->q>0) {
51 
52  if(pars->mbar<0) {

54  ret-=128;
55  }
56 
57  if(pars->kappaq<=0) {
58  fprintf(stderr,"%s: Error: kappaq must be greater than 0 if q>0\n",__func__);
59  ret-=256;
60  }
61  }
62 
63  if(pars->tmax<=0) {
64  fprintf(stderr,"%s: Error: tmax must be greater than 0\n",__func__);
65  ret-=512;
66  }
67 
68  if(pars->nstart<=0) {
69  fprintf(stderr,"%s: Error: nstart must be greater than 0\n",__func__);

200 int simulate(sim_vars* sv)
201 {
202  int i;
203  sim_pars const* sim=&(sv->pars);
204 
205  sv->ii=sv->iis;
206  sv->ii->event_time=0;
207  sv->ii->nevents=1;
208  sv->ii->curevent=0;
209  sv->ii->ninfections=sim->nstart;
210  sv->new_event_proc_func(sv);
211 
212  for(i=sim->nstart-1; i>=0; --i) {
213  DEBUG_PRINTF("initial individual %i\n",i);
214  sv->ii=sv->iis+1;
215  //Generate the communicable period appropriately
216  sv->gen_time_periods_func(sv);
217  sv->ii->nevents=gsl_ran_poisson(sv->r, sim->lambda*sv->ii->comm_period);
218  DEBUG_PRINTF("Nevents (%f*%f) is %i\n", sim->lambda, sv->ii->comm_period, sv->ii->nevents);
219 
220  //If no event for the current individual in the primary layer
221  if(!sv->ii->nevents) {
222  sv->inf_proc_func_noevent(sv->ii, sv->dataptr);
223  continue;
224  }
225 
226  sv->new_inf_proc_func(sv->ii);
227  sv->ii->curevent=0;
228 
229  for(;;) {
230  sv->ii->event_time=sv->ii->latent_period+sv->ii->comm_period*(1-gsl_rng_uniform(sv->r));
231  DEBUG_PRINTF("Event %i/%i at time %f\n",sv->ii->curevent,sv->ii->nevents,sv->ii->event_time);
232 
233  sv->ii->ninfections=gsl_ran_logarithmic(sv->r, sv->pars.p);
234 
235  if(!sv->new_event_proc_func(sv)) {
236  DEBUG_PRINTF("New event returned false\n");
237 
238  //If the events have been exhausted, go down another layer
239  if(sv->ii->curevent == sv->ii->nevents-1) {
240  sv->end_inf_proc_func(sv->ii, sv->dataptr);
241  goto done_parsing;
242  }
243 
244  //Else
245  //Move to the next event for the individual
246  ++(sv->ii->curevent);
247 
248  } else break;
249  }
250  sv->ii->curinfection=0;
251  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
252 
253  //Create a new infected individual
254  for(;;) {
255  ++(sv->ii);
256  DEBUG_PRINTF("Move to next layer (%li)\n",sv->ii-sv->iis);
257 
258  //If reaching the end of the allocated array, increase its size
259  if(sv->ii==sv->iis+sv->nlayers) {
260  sv->nlayers*=II_ARRAY_GROW_FACT;
261  DEBUG_PRINTF("Growing layers to %i\n",sv->nlayers);
262  uint64_t layer=sv->ii-sv->iis;
263  sv->iis=(infindividual*)realloc(sv->iis,sv->nlayers*sizeof(infindividual));
264  sv->increase_layers_proc_func(sv->iis+layer,sv->nlayers-layer);
265  sv->ii=sv->iis+layer;
266  }
267  //Generate the communicable period appropriately
268  sv->gen_time_periods_func(sv);
269 
270  //Generate the number of events
271  sv->ii->nevents=gsl_ran_poisson(sv->r, sim->lambda*sv->ii->comm_period);
272  DEBUG_PRINTF("Nevents (%f*%f) is %i\n", sim->lambda, sv->ii->comm_period, sv->ii->nevents);
273 
274  //If the number of events is non-zero
275  if(sv->ii->nevents) {
276  sv->ii->curevent=0;
277  sv->new_inf_proc_func(sv->ii);
278  //Generate the event time
279 gen_event:
280  sv->ii->event_time=(sv->ii-1)->event_time+sv->ii->latent_period+sv->ii->comm_period*(1-gsl_rng_uniform(sv->r));
281  DEBUG_PRINTF("Event %i/%i at time %f\n",sv->ii->curevent,sv->ii->nevents,sv->ii->event_time);
282 
283  //Generate the number of infections and the associated index for
284  //the current event
285  //Move to the next layer
286  sv->ii->ninfections=gsl_ran_logarithmic(sv->r, sv->pars.p);
287 
288  if(sv->new_event_proc_func(sv)) {
289  sv->ii->curinfection=0;
290  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
291  continue;
292 
293  } else {
294 
295  //If the events have not been exhausted
296  if(sv->ii->curevent < sv->ii->nevents-1) {
297  //Move to the next event for the individual
298  ++(sv->ii->curevent);
299  goto gen_event;
300  }
301  sv->end_inf_proc_func(sv->ii, sv->dataptr);
302  }
303 
304  } else sv->inf_proc_func_noevent(sv->ii, sv->dataptr);
305 
306  //All events for the current individual have been exhausted
307  for(;;) {
308 
309  if(sv->ii == sv->iis+1) goto done_parsing;
310  //Move down one layer
311  --(sv->ii);
312  DEBUG_PRINTF("Move to previous layer (%li)\n",sv->ii-sv->iis);
313 
314  //If the infections have been exhausted
315  if(sv->ii->curinfection == sv->ii->ninfections-1) {
316 
317  //If the events have been exhausted, go down another layer
318  if(sv->ii->curevent == sv->ii->nevents-1) {
319  sv->end_inf_proc_func(sv->ii, sv->dataptr);
320  continue;
321  }
322 
323  //Else
324  //Move to the next event for the individual
325  ++(sv->ii->curevent);
326  goto gen_event;
327  }
328 
329  //Look at the next infected individual in the current event
330  ++(sv->ii->curinfection);
331  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
332  break;
333  }
334  }
335 
336  //Else if no event for the current infected individual
337 done_parsing:
338  ;
339  }
340 
341  return 0;
