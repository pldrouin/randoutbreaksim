13 int sim_pars_check(model_pars const* pars)
14 {
15  int ret=0;
16 
17  if(pars->tbar<=0) {

43  ret-=32;
44  }

46  if(pars->pit<0) {
47  fprintf(stderr,"%s: Error: pit must be non-negative\n",__func__);
48  ret-=64;

50  } else if(pars->pit>0) {

53  fprintf(stderr,"%s: Error: itbar must be non-negative if pit>0\n",__func__);
54  ret-=128;
55  }
56 
57  if(pars->kappait<=0) {
58  fprintf(stderr,"%s: Error: kappait must be greater than 0 if pit>0\n",__func__);
59  ret-=256;
60  }
61  }
62 
63  if(pars->q<0) {
64  fprintf(stderr,"%s: Error: q must be non-negative\n",__func__);
65  ret-=512;
66 
67  } else if(pars->q>0) {
68 
69  if(pars->mbar<0) {
70  fprintf(stderr,"%s: Error: mbar must be non-negative if q>0\n",__func__);
71  ret-=1024;
72  }
73 
74  if(pars->kappaq<=0) {
75  fprintf(stderr,"%s: Error: kappaq must be greater than 0 if q>0\n",__func__);
76  ret-=2048;
77  }
78 
79  if(pars->pim<0) {
80  fprintf(stderr,"%s: Error: pim must be non-negative\n",__func__);
81  ret-=4096;
82 
83  } else if(pars->pim>0) {
84 
85  if(pars->imbar<0) {
86  fprintf(stderr,"%s: Error: imbar must be non-negative if pim>0\n",__func__);
87  ret-=8192;
88  }
89 
90  if(pars->kappaim<=0) {
91  fprintf(stderr,"%s: Error: kappaim must be greater than 0 if pim>0\n",__func__);
92  ret-=16384;
93  }
94  }
95  }
96 
97  if(pars->tmax<=0) {
98  fprintf(stderr,"%s: Error: tmax must be greater than 0\n",__func__);
99  ret-=32768;
100  }
101 
102  if(pars->nstart<=0) {
103  fprintf(stderr,"%s: Error: nstart must be greater than 0\n",__func__);
104  ret-=65536;
105  }
106  return ret;
107 }

522

585

590

594  continue;
595 
596  } else {

599  if(sv->ii->curevent < sv->ii->nevents-1) {
600  //Move to the next event for the individual
601  ++(sv->ii->curevent);
602  goto gen_event;

605  }
606 
607  } else sv->inf_proc_func_noevent(sv->ii, sv->dataptr);
608 
609  //All events for the current individual have been exhausted
610  for(;;) {
611 
612  if(sv->ii == sv->iis+1) goto done_parsing;
613  //Move down one layer
614  --(sv->ii);
615  DEBUG_PRINTF("Move to previous layer (%li)\n",sv->ii-sv->iis);
616 
617  //If the infections have been exhausted
618  if(sv->ii->curinfection == sv->ii->ninfections-1) {
619 
620  //If the events have been exhausted, go down another layer
621  if(sv->ii->curevent == sv->ii->nevents-1) {
622  sv->end_inf_proc_func(sv->ii, sv->dataptr);
623  continue;
624  }
625 
626  //Else

628  ++(sv->ii->curevent);
629  goto gen_event;
630  }
631 
632  //Look at the next infected individual in the current event
633  ++(sv->ii->curinfection);
634  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
635  break;
636  }
637  }
638 
639  //Else if no event for the current infected individual
