155 int simulate(sim_vars* sv)
156 {
157  int i;
158  model_pars const* sim=&(sv->pars);
159 
160  sv->ii=sv->iis;
161  sv->ii->event_time=0;
162  sv->ii->nevents=1;
163  sv->ii->curevent=0;
164  sv->ii->ninfections=sim->nstart;
165  sv->new_event_proc_func(sv);
166 
167  for(i=sim->nstart-1; i>=0; --i) {
168  DEBUG_PRINTF("initial individual %i\n",i);
169  sv->ii=sv->iis+1;
170  //Generate the communicable period appropriately
171  sv->gen_time_periods_func(sv);
172  sv->ii->nevents=gsl_ran_poisson(sv->r, sim->lambda*sv->ii->comm_period);
173  DEBUG_PRINTF("Nevents (%f*%f) is %i\n", sim->lambda, sv->ii->comm_period, sv->ii->nevents);
174 
175  //If no event for the current individual in the primary layer
176  if(!sv->ii->nevents) {
177  sv->inf_proc_func_noevent(sv->ii, sv->dataptr);
178  continue;
179  }
180 
181  sv->new_inf_proc_func(sv->ii);
182  sv->ii->curevent=0;
183 
184  for(;;) {
185  sv->ii->event_time=sv->ii->latent_period+sv->ii->comm_period*(1-gsl_rng_uniform(sv->r));
186  DEBUG_PRINTF("Event %i/%i at time %f\n",sv->ii->curevent,sv->ii->nevents,sv->ii->event_time);
187 
188  sv->ii->ninfections=gsl_ran_logarithmic(sv->r, sv->pars.p);
189 
190  if(!sv->new_event_proc_func(sv)) {
191  DEBUG_PRINTF("New event returned false\n");
192 
193  //If the events have been exhausted, go down another layer
194  if(sv->ii->curevent == sv->ii->nevents-1) {
195  sv->end_inf_proc_func(sv->ii, sv->dataptr);
196  goto done_parsing;
197  }
198 
199  //Else
200  //Move to the next event for the individual
201  ++(sv->ii->curevent);
202 
203  } else break;
204  }
205  sv->ii->curinfection=0;
206  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
207 
208  //Create a new infected individual
209  for(;;) {
210  ++(sv->ii);
211  DEBUG_PRINTF("Move to next layer (%li)\n",sv->ii-sv->iis);
212 
213  //If reaching the end of the allocated array, increase its size
214  if(sv->ii==sv->iis+sv->nlayers) {
215  sv->nlayers*=II_ARRAY_GROW_FACT;
216  DEBUG_PRINTF("Growing layers to %i\n",sv->nlayers);
217  uint64_t layer=sv->ii-sv->iis;
218  sv->iis=(infindividual*)realloc(sv->iis,sv->nlayers*sizeof(infindividual));
219  sv->increase_layers_proc_func(sv->iis+layer,sv->nlayers-layer);
220  sv->ii=sv->iis+layer;
221  }
222  //Generate the communicable period appropriately
223  sv->gen_time_periods_func(sv);
224 
225  //Generate the number of events
226  sv->ii->nevents=gsl_ran_poisson(sv->r, sim->lambda*sv->ii->comm_period);
227  DEBUG_PRINTF("Nevents (%f*%f) is %i\n", sim->lambda, sv->ii->comm_period, sv->ii->nevents);
228 
229  //If the number of events is non-zero
230  if(sv->ii->nevents) {
231  sv->ii->curevent=0;
232  sv->new_inf_proc_func(sv->ii);
233  //Generate the event time
234 gen_event:
235  sv->ii->event_time=(sv->ii-1)->event_time+sv->ii->latent_period+sv->ii->comm_period*(1-gsl_rng_uniform(sv->r));
236  DEBUG_PRINTF("Event %i/%i at time %f\n",sv->ii->curevent,sv->ii->nevents,sv->ii->event_time);
237 
238  //Generate the number of infections and the associated index for
239  //the current event
240  //Move to the next layer
241  sv->ii->ninfections=gsl_ran_logarithmic(sv->r, sv->pars.p);
242 
243  if(sv->new_event_proc_func(sv)) {
244  sv->ii->curinfection=0;
245  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
246  continue;
247 
248  } else {
249 
250  //If the events have not been exhausted
251  if(sv->ii->curevent < sv->ii->nevents-1) {
252  //Move to the next event for the individual
253  ++(sv->ii->curevent);
254  goto gen_event;
255  }
256  sv->end_inf_proc_func(sv->ii, sv->dataptr);
257  }
258 
259  } else sv->inf_proc_func_noevent(sv->ii, sv->dataptr);
260 
261  //All events for the current individual have been exhausted
262  for(;;) {
263 
264  if(sv->ii == sv->iis+1) goto done_parsing;
265  //Move down one layer
266  --(sv->ii);
267  DEBUG_PRINTF("Move to previous layer (%li)\n",sv->ii-sv->iis);
268 
269  //If the infections have been exhausted
270  if(sv->ii->curinfection == sv->ii->ninfections-1) {
271 
272  //If the events have been exhausted, go down another layer
273  if(sv->ii->curevent == sv->ii->nevents-1) {
274  sv->end_inf_proc_func(sv->ii, sv->dataptr);
275  continue;
276  }
277 
278  //Else
279  //Move to the next event for the individual
280  ++(sv->ii->curevent);
281  goto gen_event;
282  }
283 
284  //Look at the next infected individual in the current event
285  ++(sv->ii->curinfection);
286  DEBUG_PRINTF("Infection %i/%i\n",sv->ii->curinfection,sv->ii->ninfections);
287  break;
288  }
289  }
290 
291  //Else if no event for the current infected individual
292 done_parsing:
293  ;
294  }
295 
296  return 0;
297 }
