244 int config_solve_R0_group(sim_pars* pars)
245 {
246  //If p is provided as an input
247  if(!isnan(pars->p)) {
248 
249  if(pars->p<0) {
250  fprintf(stderr,"%s: Error: p must be non-negative\n",__func__);
251  return -1;
252  }
253  pars->mu=(pars->p>0?-pars->p/((1-pars->p)*log(1-pars->p)):1);
254  }
255 
256  if(!isnan(pars->tbar) && pars->tbar<=0) {
257  fprintf(stderr,"%s: Error: tbar must be greater than 0\n",__func__);
258  return -2;
259  }
260 
261  if(!isnan(pars->lambda) && pars->lambda<=0) {
262  fprintf(stderr,"%s: Error: lambda must be greater than 0\n",__func__);
263  return -3;
264  }
265 
266  if(!isnan(pars->R0) && pars->R0<=0) {
267  fprintf(stderr,"%s: Error: R0j must be greater than 0\n",__func__);
268  return -4;
269  }
270 
271  //Solve for the missing parameter
272  if(isnan(pars->R0)) pars->R0=pars->lambda*pars->tbar*pars->mu;
273 
274  else if(isnan(pars->lambda)) pars->lambda=pars->R0/(pars->tbar*pars->mu);
275 
276  else if(isnan(pars->tbar)) pars->tbar=pars->R0/(pars->lambda*pars->mu);
277 
278  else pars->mu=pars->R0/(pars->lambda*pars->tbar);
279 
280  //If p is unknown solve for it numerically
281  if(isnan(pars->p)) {
282 
283  if(pars->mu > 1) {
284  root_finder* rf=root_finder_init(logroot, &pars->mu);
285  pars->p=0.999;
286 
287  int ret=root_finder_find(rf, RF_P_EPS, 100, RF_P_EPS, 1-RF_P_EPS, &pars->p);
288 
289  root_finder_free(rf);
290 
291  if(ret) return ret;
292 
293  } else pars->p=0;
294  }
295  return 0;
296 }

298 int config_solve_gamma_group(double* ave, double* kappa, double* x95)
299 {
300  if(!(*ave>=0)) {
301  fprintf(stderr,"%s: Error: The average of the distributio must be non-negative.\n",__func__);
302  return -1;
303  }
304 
305  if(isnan(*x95)) {
306 
307  if(!(*kappa>0)) {
308  fprintf(stderr,"%s: Error: The kappa parameter of the distributio must be non-negative.\n",__func__);
309  return -1;
310  }
311 
312  if(*kappa != INFINITY) {
313  double pars[2]={*ave * *kappa, *kappa};
314  root_finder* rf=root_finder_init(gpercroot, pars);
315  *x95=*ave;
316 
317  int ret=root_finder_find(rf, RF_GPERC_EPSF, 100, *ave, 1e100, x95);
318 
319  root_finder_free(rf);
320 
321  if(ret) return ret;
322 
323  } else *x95 = *ave;
324 
325  } else {
326 
327  if(!(*x95>=*ave)) {
328  fprintf(stderr,"%s: Error: The 95th percentile of the distributio cannot be smaller than the average\n",__func__);
329  return -1;
330  }
331 
332  if(*x95 != *ave) {
333  *kappa=1;
334  const double otherkappa=*kappa*0.9;
335  double pars[4]={*ave, *x95, otherkappa, gpercrootfunc(*ave * otherkappa, *x95 * otherkappa)};
336  root_finder* rf=root_finder_init(gkapparoot, pars);
337 
338  int ret=root_finder_find(rf, RF_GKAPPA_EPSF, 100, 1e-100, 1e100, kappa);
339 
340  root_finder_free(rf);
341 
342  if(ret) return ret;
343 
344  } else *kappa = INFINITY;
345  }
346  return 0;
