293 /**
294  * @brief Computes g_ave (truncated Gaussian distribution) from the mean and the
295  * standard deviation.
296  */
297 inline static double gauss_trunc_g_ave(const double mu, const double sigma)
298 {
299  int32_t i;
300  int32_t mui=floor(mu+0.5);
301  const double dmu=mu-mui;
302  const double psmu=dmu-0.5;
303  const double nsmu=dmu+0.5;
304  int32_t nbins=mui-2;
305  int32_t lasti=0;
306  double lastrangeint;
307  double newrangeint;
308  double lastint=gsl_cdf_ugaussian_P((-nbins-1-psmu)/sigma);;
309  double newint;
310  double mean=0;
311  double fint=0;
312  double dint;
313  double meanbuf;
314  double fintbuf;
315 
316  for(i=-nbins; i<=0; ++i) {
317  //printf("i: %i, lastint: %22.15e, fint: %22.15e, mean: %22.15e\n", i, lastint, fint, mean);
318 
319  newint=gsl_cdf_ugaussian_P((i-psmu)/sigma);
320  dint=newint-lastint;
321  //printf("dint: %22.15e\n",dint);
322  fint+=dint;
323  mean+=dint*i;
324  lastint=newint;
325  }
326  lastrangeint=lastint;
327 
328  for(;;) {
329  fintbuf=0;
330  meanbuf=0;
331  newrangeint=lastint=gsl_cdf_ugaussian_P((lasti+nbins-psmu)/sigma);
332 
333  for(i=lasti+nbins; i>lasti+1; --i) {
334  //printf("i: %i, lastint: %22.15e, fintbuf: %22.15e, meanbuf: %22.15e\n", i, lastint, fintbuf, meanbuf);
335  newint=gsl_cdf_ugaussian_P((i-nsmu)/sigma);
336  dint=lastint-newint;
337  //printf("dint: %22.15e\n",dint);
338  fintbuf+=dint;
339  meanbuf+=dint*i;
340  lastint=newint;
341  }
342  //printf("i: %i, lastint: %22.15e, fintbuf: %22.15e, meanbuf: %22.15e\n", lasti+1, lastint, fintbuf, meanbuf);
343  dint=lastint-lastrangeint;
344  //printf("dint: %22.15e\n",dint);
345  fintbuf+=dint;
346  meanbuf+=dint*(lasti+1);
347 
348  if(newrangeint==1 || (fint+fintbuf == fint && mean+meanbuf == mean)) break;
349  fint+=fintbuf;
350  mean+=meanbuf;
351  lastrangeint=newrangeint;
352  lasti+=nbins;
353  nbins*=2;
354  //printf("nbins to %i\n",nbins);
355  }
356 
357  return mean/fint+mui;
358 }
