197 /**
198  * @brief Computes x95 for a gamma cumulative distribution using Newton's
199  * method.
200  *
201  * this function estimates the value of x95 (the x value corresponding to the
202  * 95th percentile) for a gamma distribution, using Newton's method.
203  *
204  * @param x: Current value for x95 (input/output)
205  * @param diff: Current value for the discrepancy between the CDF evaluated at
206  * the current x95 parameter value and the 95th percentile. (output)
207  * @param params: Pointer to an array of parameters. First parameter is the a parameter
208  * for the gamma distribution and the second parameters is kappa. (input)
209  */
210 inline static void gpercroot(double* x, double* diff, void* params){const double t=*(((double*)params)+1)* *x; *diff=gpercrootfunc(*(double*)params,t); *x-=*diff * tgamma(*(double*)params) / (pow(t,*(double*)params-1)*exp(-t) * *(((double*)params)+1));}
