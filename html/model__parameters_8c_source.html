148  fprintf(stderr,"%s: Error: The pinf parameter must have a value in the interval (0,1]\n",__func__);
149  return -2;
150  }
151

154  return -3;
155  }
156

158  fprintf(stderr,"%s: Error: lambda must be greater than 0\n",__func__);
159  return -4;
160  }
161 
162  if(!isnan(pars->R0) && pars->R0<=0) {
163  fprintf(stderr,"%s: Error: R0j must be greater than 0\n",__func__);
164  return -5;
165  }
166 
167  //If g_ave, p or mu is provided as an input
168  if(!isnan(pars->g_ave) || !isnan(pars->p) || !isnan(pars->mu)) {
169 
170  //If g_ave is provided
171  if(!isnan(pars->g_ave)) {
172 
173  if(!(pars->g_ave>=2)) {
174  fprintf(stderr,"%s: Error: g_ave must be greater than or equal to 2\n",__func__);
175  return -1;
176  }
177 
178  if(pars->grouptype&ro_group_log_attendees_plus_1) {
179  pars->mu=pars->g_ave-1;
180  //Solve for p numerically from mu
181  ret=model_solve_p_from_mu(pars);
182 
183  if(ret) return ret;
184 
185  } else {
186  //Solve for p numerically from g_ave
187  ret=model_solve_p_from_mean(pars->g_ave,pars);
188 
189  if(ret) return ret;
190  pars->mu=-pars->p/((1-pars->p)*log(1-pars->p));
191  }
192 
193  } else {
194 
195  //Else if p is provided
196  if(!isnan(pars->p)) {
197 
198  if(!(pars->p>=0) || !(pars->p<1)) {
199  fprintf(stderr,"%s: Error: p must be non-negative and smaller than 1\n",__func__);
200  return -1;
201  }
202  pars->mu=(pars->p>0?-pars->p/((1-pars->p)*log(1-pars->p)):1);
203 
204  //Else if it is mu that is provided
205  } else {
206 
207  if(!(pars->mu>=1)) {
208  fprintf(stderr,"%s: Error: mu must be greater than or equal to 1\n",__func__);
209  return -1;
210  }
211  //Solve for p numerically from mu
212  ret=model_solve_p_from_mu(pars);
213 
214  if(ret) return ret;
215  }
216

218 
219  else if(pars->p == 0) pars->g_ave=2;
220

223  pars->g_ave=-pars->p*pars->p/((1-pars->p)*(l1mp+pars->p));
224  }
225  }
226

228

232

236

254 
255  if(ret) return ret;

261 
262  if(ret) return ret;

321 int model_solve_gamma_group(double* ave, double* kappa, double* x95)
322 {
323  if(!(*ave>=0)) {
324  fprintf(stderr,"%s: Error: The average of the distribution must be non-negative.\n",__func__);
325  return -1;
326  }
327 
328  if(isnan(*x95)) {
329 
330  if(!(*kappa>=1/ *ave)) {
331  fprintf(stderr,"%s: Error: The kappa parameter of the distribution must have a value greater than or equal to 1/ave.\n",__func__);
332  return -1;
333  }
334 
335  if(*kappa != INFINITY) {
336  double pars[2]={*ave * *kappa, *kappa};
337  root_finder* rf=root_finder_init(gpercroot, pars);
338  *x95=*ave;
339 
340  int ret=root_finder_find(rf, RF_GPERC_EPSF, 100, *ave, 1e100, x95);
341 
342  root_finder_free(rf);
343 
344  if(ret) {
345 
346  if(ret==-3) fprintf(stderr,"%s: Warning: Convergence seems to have been reached, but the root discrepancy is larger than provided!\n",__func__);
347 
348  else if(ret==-2) {
349  fprintf(stderr,"%s: Error: Root could not be found!\n",__func__);
350  return ret;
351  }
352  }
353 
354  } else *x95 = *ave;
355 
356  } else {
357 
358  if(!(*x95>=*ave)) {

360  return -1;
361  }
362

370 
371  root_finder_free(rf);
372 
373  if(ret) {

376 
377  else if(ret==-2) {

379  return ret;
