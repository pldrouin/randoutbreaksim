140 int model_solve_R0_group(model_pars* pars)
141 {
142  //If p is provided as an input
143  if(!isnan(pars->p)) {
144 
145  if(pars->p<0) {
146  fprintf(stderr,"%s: Error: p must be non-negative\n",__func__);
147  return -1;
148  }
149  pars->mu=(pars->p>0?-pars->p/((1-pars->p)*log(1-pars->p)):1);
150  }
151 
152  if(!isnan(pars->tbar) && pars->tbar<=0) {
153  fprintf(stderr,"%s: Error: tbar must be greater than 0\n",__func__);
154  return -2;
155  }
156 
157  if(!isnan(pars->lambda) && pars->lambda<=0) {
158  fprintf(stderr,"%s: Error: lambda must be greater than 0\n",__func__);
159  return -3;
160  }
161 
162  if(!isnan(pars->R0) && pars->R0<=0) {
163  fprintf(stderr,"%s: Error: R0j must be greater than 0\n",__func__);
164  return -4;
165  }
166 
167  //Solve for the missing parameter
168  if(isnan(pars->R0)) pars->R0=pars->lambda*pars->tbar*pars->mu;
169 
170  else if(isnan(pars->lambda)) pars->lambda=pars->R0/(pars->tbar*pars->mu);
171 
172  else if(isnan(pars->tbar)) pars->tbar=pars->R0/(pars->lambda*pars->mu);
173 
174  else pars->mu=pars->R0/(pars->lambda*pars->tbar);
175 
176  //If p is unknown solve for it numerically
177  if(isnan(pars->p)) {
178 
179  if(pars->mu > 1) {
180  root_finder* rf=root_finder_init(logroot, &pars->mu);
181  pars->p=0.999;
182 
183  int ret=root_finder_find(rf, RF_P_EPSF, 100, RF_P_EPSF, 1-RF_P_EPSF, &pars->p);
184 
185  root_finder_free(rf);
186 
187  if(ret) return ret;
188 
189  } else pars->p=0;
190  }
191  return 0;

195 {

221  } else {
