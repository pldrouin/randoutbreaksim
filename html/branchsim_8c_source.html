37 
38  sv->gen_time_origin_func(sv);

40 
41  DEBUG_PRINTF("Comm period is %f%s\n",sv->brsim.iis[1].comm_period,(sv->brsim.iis[1].commpertype&ro_commper_tmax?" (reached end)":"")); \
42

49 
50  sv->curii->nevents=gsl_ran_poisson(sv->r, sim->lambda*sv->curii->comm_period);
51  DEBUG_PRINTF("Nevents (%f*%f) is %i\n", sim->lambda, sv->curii->comm_period, sv->curii->nevents);
52

54  if(!sv->curii->nevents) {
55  sv->inf_proc_func_noevent(sv->curii, sv->dataptr);
56  continue;
57  }
58 
59  sv->curii->curevent=0;
60 
61  for(;;) {

63  //sv->curii->event_time=sv->curii->latent_period+sv->curii->comm_period*rng_rand_pu01d((rng_stream*)sv->r->state);
64  DEBUG_PRINTF("Event %i/%i at time %f\n",sv->curii->curevent,sv->curii->nevents,sv->curii->event_time);
65 
66  //sv->curii->ninfections=gsl_ran_logarithmic(sv->r, sv->pars.p);
67  sv->gen_att_inf_func(sv);
68  DEBUG_PRINTF("%u attendees and %u infections were generated\n",sv->curii->nattendees,sv->curii->ninfections);
69 
70  if(sv->curii->nattendees<2 || !sv->new_event_proc_func(sv)) {
71  DEBUG_PRINTF("New event returned false\n");
72 
73  //If the events have been exhausted, go down another layer
74  if(sv->curii->curevent == sv->curii->nevents-1) {
75  sv->end_inf_proc_func(sv->curii, sv->dataptr);
76  goto done_parsing;
77  }
78 
79  //Else
80  //Move to the next event for the individual
81  ++(sv->curii->curevent);
82 
83  } else break;
84  }
85  sv->curii->curinfection=0;
86  DEBUG_PRINTF("Infection %i/%i\n",sv->curii->curinfection,sv->curii->ninfections);
87 
88  //Create a new infected individual
89  for(;;) {

92

98  sv->brsim.iis=(infindividual*)realloc(sv->brsim.iis,sv->brsim.nlayers*sizeof(infindividual));
99 
100  for(uint32_t i=sv->brsim.nlayers-1; i>=layer; --i) sv->ii_alloc_proc_func(sv->brsim.iis+i);
101  sv->curii=sv->brsim.iis+layer;
102  }
103  //Generate the communicable period appropriately
104  sv->gen_time_periods_func(sv, sv->curii, sv->curii-1, (sv->curii-1)->event_time);
105  sv->curii->commpertype|=ro_commper_tmax*(sv->curii->end_comm_period > sv->pars.tmax);
106  DEBUG_PRINTF("Comm period is %f%s\n",sv->curii->comm_period,(sv->curii->commpertype&ro_commper_tmax?" (reached end)":"")); \
107 
108  //Generate the number of events
109  sv->curii->nevents=gsl_ran_poisson(sv->r, sim->lambda*sv->curii->comm_period);
110  DEBUG_PRINTF("Nevents (%f*%f) is %i\n", sim->lambda, sv->curii->comm_period, sv->curii->nevents);
111 
112  //If the number of events is non-zero
113  if(sv->curii->nevents) {
114  sv->curii->curevent=0;
115  sv->new_inf_proc_func(sv, sv->curii);
116  //Generate the event time
117 gen_event:
118  sv->curii->event_time=sv->curii->end_comm_period-sv->curii->comm_period*gsl_rng_uniform(sv->r);
119  //sv->curii->event_time=(sv->curii-1)->event_time+sv->curii->latent_period+sv->curii->comm_period*rng_rand_pu01d((rng_stream*)sv->r->state);
120  DEBUG_PRINTF("Event %i/%i at time %f\n",sv->curii->curevent,sv->curii->nevents,sv->curii->event_time);
121 
122  //Generate the number of infections and the associated index for
123  //the current event
124  //Move to the next layer
125  //sv->curii->ninfections=gsl_ran_logarithmic(sv->r, sv->pars.p);
126  sv->gen_att_inf_func(sv);
127  DEBUG_PRINTF("%u attendees and %u infections were generated\n",sv->curii->nattendees,sv->curii->ninfections);
128 
129  if(sv->curii->nattendees>1 && sv->new_event_proc_func(sv)) {
130  sv->curii->curinfection=0;
131  DEBUG_PRINTF("Infection %i/%i\n",sv->curii->curinfection,sv->curii->ninfections);
132  continue;
133 
134  } else {
135 
136  //If the events have not been exhausted
137  if(sv->curii->curevent < sv->curii->nevents-1) {
138  //Move to the next event for the individual
139  ++(sv->curii->curevent);
140  goto gen_event;
141  }
142  sv->end_inf_proc_func(sv->curii, sv->dataptr);
143  }
144 
145  } else sv->inf_proc_func_noevent(sv->curii, sv->dataptr);
146 
147  //All events for the current individual have been exhausted
148  for(;;) {
149 
150  if(sv->curii == sv->brsim.iis+1) goto done_parsing;
151  //Move down one layer
152  --(sv->curii);
153  DEBUG_PRINTF("Move to previous layer (%li)\n",sv->curii-sv->brsim.iis);
154 
155  //If the infections have been exhausted
156  if(sv->curii->curinfection == sv->curii->ninfections-1) {
157 
158  //If the events have been exhausted, go down another layer
159  if(sv->curii->curevent == sv->curii->nevents-1) {
160  sv->end_inf_proc_func(sv->curii, sv->dataptr);
161  continue;
162  }
163 
164  //Else
165  //Move to the next event for the individual
166  ++(sv->curii->curevent);
167  goto gen_event;
168  }
169 
170  //Look at the next infected individual in the current event
171  ++(sv->curii->curinfection);
172  DEBUG_PRINTF("Infection %i/%i\n",sv->curii->curinfection,sv->curii->ninfections);
173  break;
174  }
175  }
176 
177  //Else if no event for the current primary infected individual
178 done_parsing:
179  ;
180  }
181 
182  return 0;
