176 /**
177  * @brief Process the statistics after the last transmission event for an
178  * infectious individual that participates to some transmission events.
179  *
180  * After an infectious infectious individual participated to its last
181  * transmission event, this function adds the total number of infections from
182  * the individual to the R sum, the individual's communicable period to the sum
183  * of communicable periods and the number of transmission events this individual
184  * participated to to the sum of events. If the individual was still infectious
185  * at tmax, then the path is set to not go extinct. Otherwise, the
186  * extinction time for the generating path is updated if its current value was
187  * exceeded, using the current event time for the individual's parent and the
188  * individual's communicable period. The timeline for the number of infectious
189  * individuals is also updated.
190  *
191  * @param inf: Pointer to the infectious individual.
192  * @param ptr: Pointer to the summary statistics.
193  * */
194 inline static void std_stats_end_inf(infindividual* inf, void* ptr)
195 {
196  DEBUG_PRINTF("Number of infections was %u\n",*(uint32_t*)inf->dataptr);
197  ((std_summary_stats*)ptr)->rsum+=*(uint32_t*)inf->dataptr;
198  ((std_summary_stats*)ptr)->commpersum+=inf->comm_period;
199 #ifdef NUMEVENTSSTATS
200  ((std_summary_stats*)ptr)->neventssum+=inf->nevents;
201 #endif
202 
203  const double inf_end=(inf-1)->event_time+inf->comm_period;
204 
205  //If truncated by tmax
206  if(inf->infectious_at_tmax) ((std_summary_stats*)ptr)->extinction=false;
207 
208  else {
209  //++((std_summary_stats*)ptr)->n_ended_infections;
210 
211  if(inf_end > ((std_summary_stats*)ptr)->extinction_time) ((std_summary_stats*)ptr)->extinction_time=inf_end;
212  }
213 
214  int i;
215  const int end_comm_per=(inf_end >= ((std_summary_stats*)ptr)->npers ? ((std_summary_stats*)ptr)->npers-1 : (int)inf_end);
216 
217  for(i=(int)((inf-1)->event_time); i<=end_comm_per; ++i) ++(((std_summary_stats*)ptr)->inf_timeline[i]);
218 }
